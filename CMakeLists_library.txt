# CMakeLists.txt cho Library - Có thể được sử dụng bởi project khác
cmake_minimum_required(VERSION 3.10)

project(KaraokeScorer VERSION 1.0.0 LANGUAGES CXX)

# C++ Standard
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# --- Find Python ---
find_package(Python3 COMPONENTS Interpreter Development QUIET)
if(Python3_FOUND)
    set(PYTHON_INCLUDE_DIRS ${Python3_INCLUDE_DIRS})
    set(PYTHON_LIBRARIES ${Python3_LIBRARIES})
    message(STATUS "Found Python3: ${Python3_EXECUTABLE}")
else()
    find_package(PythonLibs REQUIRED)
    set(PYTHON_INCLUDE_DIRS ${PYTHON_INCLUDE_DIRS})
    set(PYTHON_LIBRARIES ${PYTHON_LIBRARIES})
    message(STATUS "Found Python (legacy): ${PYTHON_INCLUDE_DIRS}")
endif()

# --- Create Library Target ---
# Tạo một header-only library hoặc static library
add_library(KaraokeScorer STATIC
    KaraokeScorer.cpp
)

# --- Include Directories ---
# Public include directory - các project khác sẽ tự động có access
target_include_directories(KaraokeScorer
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}  # Để include KaraokeScorer.h
        ${PYTHON_INCLUDE_DIRS}
)

# --- Link Libraries ---
target_link_libraries(KaraokeScorer
    PUBLIC
        ${PYTHON_LIBRARIES}
)

# --- Install Configuration (Optional) ---
# Cho phép install library để dùng với find_package
include(GNUInstallDirs)

install(TARGETS KaraokeScorer
    EXPORT KaraokeScorerTargets
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    INCLUDES DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
)

install(FILES KaraokeScorer.h
    DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
)

install(EXPORT KaraokeScorerTargets
    FILE KaraokeScorerTargets.cmake
    NAMESPACE KaraokeScorer::
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/KaraokeScorer
)

# --- Copy Python files to build directory ---
# Để các file Python có sẵn khi chạy
configure_file(
    ${CMAKE_CURRENT_SOURCE_DIR}/library_interface.py
    ${CMAKE_CURRENT_BINARY_DIR}/library_interface.py
    COPYONLY
)

configure_file(
    ${CMAKE_CURRENT_SOURCE_DIR}/pitch_extractor.py
    ${CMAKE_CURRENT_BINARY_DIR}/pitch_extractor.py
    COPYONLY
)

configure_file(
    ${CMAKE_CURRENT_SOURCE_DIR}/pitch_matcher.py
    ${CMAKE_CURRENT_BINARY_DIR}/pitch_matcher.py
    COPYONLY
)

# --- Install Python files ---
install(FILES
    library_interface.py
    pitch_extractor.py
    pitch_matcher.py
    DESTINATION ${CMAKE_INSTALL_DATADIR}/KaraokeScorer/python
)
